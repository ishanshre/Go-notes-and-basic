# Go single line comment
    Single line comments starts with two forward slashes ( // )
    ```
    package main
    import ("fmt")
    func main() {
        // This is a comment
        fmt.Println("Hello World!")
    }

# Go Multi Line comment
    Multi line comment starts with ( /* ) and ends with ( */ )

    ```
    package main
    import ("fmt")
    func main() {
        /*
        This is multi line commnet
        Hello World
        */
        fmt.Println("Hello World!")
    }
